# application.yml
spring:
  mvc:
    # 静态资源映射
    static-path-pattern: /static/**
    
    # 日期时间格式化
    format:
      date: yyyy-MM-dd
      date-time: yyyy-MM-dd HH:mm:ss
      time: HH:mm:ss
    
    # 跨域配置
    cors:
      allowed-origins: "http://localhost:8080"
      allowed-methods: GET,POST,PUT,DELETE
    
    # 路径匹配策略
    pathmatch:
      matching-strategy: ant_path_matcher
  
  # 文件上传配置
  servlet:
    multipart:
      max-file-size: 10MB
      max-request-size: 10MB
  
  # Jackson JSON 配置
  jackson:
    date-format: yyyy-MM-dd HH:mm:ss
    time-zone: GMT+8
    default-property-inclusion: non_null

# 服务器配置
server:
  port: 8080
  servlet:
    context-path: /api
    encoding:
      charset: UTF-8
      enabled: true
      force: true
